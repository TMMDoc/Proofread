Program Sttructure

Program Sttructure

flowchart TD
    START([Start]) --> INIT

    subgraph MAIN["Main Program: 200x200_part.LS"]
        direction TB
        INIT["LBL[100] Initialize\nCall Initialize · Reset IOs\nDO[53:ProgramStart]=ON\nR[15:SPD_J]=50  R[16:SPD_L]=1000\nR[17:BendNumStatus]=1  R[10:ProgramState]=1"]
        INIT --> SM

        SM{"LBL[110]\nProgram State Machine\nSELECT R[10]"}
        SM -- "= 1" --> S1
        SM -- "= 2" --> S2
        SM -- "= 3" --> S3
        SM -- "= 4" --> S4
        SM -- "ELSE" --> SM

        S1["LBL[1] PartPickupInit"]
        S1 --> CND1{"FirstBendActive\nAND R[19]≠1?"}
        CND1 -- Yes --> PCALL["Call Sub_Pickup\nCall Sub_CenteringTable\nR[19]=1"]
        PCALL --> CND2
        CND1 -- No --> CND2
        CND2{"MoveToUDP active OR\nFirstBend AND !UDP_Reached?"}
        CND2 -- Yes --> R10_2["R[10]=2"] --> SM
        CND2 -- No --> CND3{"NCStart AND\nPressOk?"}
        CND3 -- Yes --> R10_3["R[10]=3"] --> SM
        CND3 -- No --> SM

        S2["LBL[2] RAMUP\nDO[51:MoveToUDP]=ON\nWAIT UDP_Reached\nDO[51:MoveToUDP]=OFF\nR[10]=1"]
        S2 --> SM

        S3["LBL[3] FASTCLOSING"]
        S3 --> CND4{"UDP_Reached?"}
        CND4 -- No --> SM
        CND4 -- Yes --> BGWAIT["DO[52:StepChange]=ON\nWAIT Backgauge in Position\nDO[55:RoboUnsafe]=ON"]
        BGWAIT --> BSEL{"SELECT R[17]\n(Bend Number)"}
        BSEL -- "= 1" --> B1CALL["CALL Bend1Positioning_Sub\n(Regrip Bend)"]
        BSEL -- "= 2" --> B2CALL["CALL Bend2Positioning_Sub\n(No-Regrip Bend)"]
        BSEL -- "ELSE" --> S4
        B1CALL --> LASTB
        B2CALL --> LASTB
        LASTB{"Last Bend Active\nDI[52:ECAT_LastBendActive]?"}
        LASTB -- No --> S3
        LASTB -- Yes --> S4

        S4["LBL[4] TODEPOSIT\nCall Sub_Deposit\nR[10]=1  R[19]=0"]
        S4 --> INIT
    end

    subgraph PICKUP["Sub_Pickup.LS"]
        direction TB
        PU1["R[50]=1  Station: Pickup\nSet UFRAME=AR[1]  UTOOL=3"]
        PU1 --> PU2["Move to Approach (J)"]
        PU2 --> PU3["Update Contact Point PR[63]\nCall FUNCCombiGripper — Activate"]
        PU3 --> PU4["Call Sub_StackSearch\nFind top of pallet stack"]
        PU4 --> PU5["SPD_L=1000  Wait 0.25s\nSeparate → LiftClear → PickupTransitionSwing"]
    end

    subgraph B1POS["Bend1Positioning_Sub.LS — Regrip Bend"]
        direction TB
        B1_1["R[50]=3  Station: Bending"]
        B1_1 --> B1CHK{"Ram Moving?\nDI[53:ECAT_RamMoving]"}
        B1CHK -- Yes --> B1WAIT["WAIT !RamMoving"] --> B1CHK
        B1CHK -- No --> B1_2["PartInsert → PartInsert2 → MutePoint\nRecord LPOS → PR[78:LPOS_MUT]"]
        B1_2 --> B1_3["DO[55:RoboUnsafe]=OFF\nDO[54:EnableRamDown]=ON\nCall BG_ALIGN(6.5, 3.855, 2 BGs)\nWAIT BelowMutePoint"]
        B1_3 --> B1_4["LBL[5] CLAMPING\nDO[49:ClampingPointBit]=ON\nWAIT ClampingPointReach"]
        B1_4 --> B1_5["LBL[6] MOVETOLDP\nDO[54:EnableRamDown]=OFF\nR[17]=2\nCall 0_Bend1Sub(10 sync pts)"]
        B1_5 --> B1_6["Regrip Sequence:\nRelease Gripper → Swivel1 → Swivel2\n→ Contact → Re-Grip  Wait 1s"]
        B1_6 --> B1_7["Reset IOs\nDO[51:MoveToUDP]=ON\nWAIT UDP_Reached\nDO[51:MoveToUDP]=OFF"]
        B1_7 --> B1_8["Decompress → Opening\n→ PostBendSafe → PostBendSafe2\nR[17]=2"]
    end

    subgraph BEND1S["BEND1SUB.LS — Synchronized Bending"]
        direction TB
        BS1["R[77]=AR[1]  Load N sync points\nFOR loop: PR[i]=P[i]\nSet SPD_L=2000"]
        BS1 --> BS2["Release Gripper\nMove to P[1] (1st sync point)\nRe-Grip  Wait 1s"]
        BS2 --> BS3["Init skip condition R[35]≠R[34]\nDO[50:MoveToLDP]=ON\nRecord Fast Clock R[36]"]
        BS3 --> BS4{"DI[56] Decompress\nReached?"}
        BS4 -- No --> BS5["Dynamic Sync Motion\nL PR[R[35]] 1msec CNT100 ACC80\n(Skip if R[35]≠R[34])"]
        BS5 --> BS4
        BS4 -- Yes --> BS6["LBL[3]: Move to last sync position\nRecord timing  Pause"]
    end

    subgraph B2POS["Bend2Positioning_Sub.LS — No-Regrip Bend"]
        direction TB
        B2_1["R[50]=3  Station: Bending"]
        B2_1 --> B2CHK{"Ram Moving?\nDI[53:ECAT_RamMoving]"}
        B2CHK -- Yes --> B2WAIT["WAIT !RamMoving"] --> B2CHK
        B2CHK -- No --> B2_2["PreBendSafe → StationFront\n→ PartInsert → PartInsert2 → MutePoint\nRecord LPOS → PR[78:LPOS_MUT]"]
        B2_2 --> B2_3["DO[55:RoboUnsafe]=OFF\nDO[54:EnableRamDown]=ON\nCall BG_ALIGN(6.5, 3.855, 3 BGs)\nWAIT BelowMutePoint"]
        B2_3 --> B2_4["LBL[5] CLAMPING\nDO[49:ClampingPointBit]=ON\nWAIT ClampingPointReach"]
        B2_4 --> B2_5["LBL[6] MOVETOLDP\nDO[54:EnableRamDown]=OFF\nR[17]=3\nCall 0_Bend2Sub(10 sync pts)"]
        B2_5 --> B2_6["Reset IOs\nDO[51:MoveToUDP]=ON\nWAIT UDP_Reached\nDO[51:MoveToUDP]=OFF"]
        B2_6 --> B2_7["Decompress → Opening\n→ PostBendSafe → PostBendSafe2\nR[17]=3"]
    end

    subgraph DEPOSIT["Sub_Deposit.LS"]
        direction TB
        D1["R[50]=4  Station: Deposit\nSet UFRAME=AR[1]"]
        D1 --> D2["Move to DepositApproach (J)\n→ Deposit position (L)"]
        D2 --> D3["Call FUNCCombiGripper — Release part\nWAIT 0.5s"]
        D3 --> D4["Move: Clear → DepositTransitionSwing (J)"]
        D4 --> D5["Reset All IOs\nR[19]=0  R[15:SPD_J]=100\nMove to Home"]
    end

    PCALL -. "Call" .-> PU1
    B1CALL -. "Call" .-> B1_1
    B2CALL -. "Call" .-> B2_1
    B1_5 -. "Call" .-> BS1
    S4 -. "Call" .-> D1
